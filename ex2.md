# Write a program to generate the following output. Use an appropiate loop. 
output :

          *        
	     ***
		   *****
		  *******
		 *********
#
---

[looping](courseid-5,lessonid-16) , [looping](courseid-5,lessonid-16)

------
# Solution

	#include <iostream>
	using namespace std;

	int main () {
		int i , j ,row = 5;
		
		for (i = 1; i <= row; i++){
			
			for (j = 1; j <= row - i; j++)
				cout << " ";
				
			for(j = 1; j <= 2 * i - 1; j++)
				cout << "*";
				
		cout << endl;
		}

 	  return 0;
	}
